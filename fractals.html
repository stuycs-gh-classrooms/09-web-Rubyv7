Kochcurve

The edited Koch_Curve randomly picks between pink or blue lines and also increases width by multiplying it by a random number created in the beginning. 

def koch_curve2(t, depth, size, width):
    x = (random.randrange(2)) + 1
    y = (random.randrange(2)) + 1
    t.width(width)
    if (depth == 1):
        t.fd(size)
    else:
        c = random.randrange(10)
        if c < 4:
                t.pencolor("pink")
        else:
                t.pencolor("blue")
        koch_curve2(t, depth-1, size, width * x)
        t.lt(60)
        koch_curve2(t, depth-1, size, width * y)
        t.rt(120)
        koch_curve2(t, depth-1, size, width * x)
        t.lt(60)
        koch_curve2(t, depth-1, size, width * y)

Sierpinski Triangle

The edited Sierpinski triangle has the top and bottom left triangles missing their bottom side, making a rhombus and another weird quadrilateral in the center. It's also purple. 

def triangle(t, size):
    t.lt(60)
    t.fd(size)
    t.rt(120)
    t.fd(size)
    t.pu()
    t.rt(120)
    t.fd(size)
    t.rt(180)
    t.pd()

def sierpinski2(t, depth, size, scale_factor=1):
    if depth == 1:
        triangle(t, size)
    else:
        t.pencolor("purple")
        sierpinski2(t, depth-1, size/2)
        t.fd(size/2)
        sierpinski2(t, depth-1, size/2)
        t.bk(size/2)
        t.lt(60)
        t.fd(size/2)
        t.pencolor("green")
        t.rt(60)
        sierpinski2(t, depth-1, size/2)
        t.rt(120)
        t.fd(size/2)
        t.lt(120)
